        function fib1(n) {
      if (n < 2) return n;
      return fib1(n-1) + fib1(n-2);

    function fib2(n) {
      var loop = (n, prev, cur) => {
        if (n === 0) return prev;
        return loop(n - 1, cur, prev + cur);
      };
      return loop(n, 0, 1);
    };
